canvas <- function(mn = 0, mx = 100) {
    plot(1:2, xlim = c(mn, mx), ylim = c(mn, mx), xaxt = "n", yaxt = "n", xlab = "", 
        ylab = "", type = "n", asp = 1)

}

drawCircle <- function(x, y, r, col) {
    angs <- seq(0, 2 * pi, length = 20)
    xpts <- x + r * cos(angs)
    ypts <- y + r * sin(angs)
    polygon(xpts, ypts, col = col, border = FALSE)
}
drawCircle(22, 23, 2, "red")
